  58         /* Sorterar resultat per kategori för snygg visning */
  59         function getBlogsByCategory($userId, $limit = 0)
  60         {
  61                 global $db;
  62 
  63                 if (!is_numeric($userId) || !is_numeric($limit)) return false;
  64 
  65                 $q  = 'SELECT t1.*,t2.categoryName,t2.categoryPermissions FROM tblBlogs AS t1';
  66                 $q .= ' LEFT JOIN tblCategories AS t2 ON (t1.categoryId=t2.categoryId AND t2.categoryType='.CATEGORY_BLOG.')';
  67                 $q .= ' WHERE t1.userId='.$userId.' AND t1.deletedBy=0';
  68 
  69                 /* Return order: First blogs categorized in global categories, then blogs categorized in user's categories, then uncategorized blogs */
  70                 $q .= ' ORDER BY t2.categoryPermissions DESC, t1.categoryId ASC, t1.timeCreated DESC';
  71                 if ($limit) $q .= ' LIMIT 0,'.$limit;
  72 
  73                 return $db->getArray($q);
  74         }

 106         /* Returns all blogs from $userId for the specified month */
 107         function getBlogsByMonth($userId, $month, $year, $order_desc = true)
 108         {
 109                 global $db;
 110 
 111                 if (!is_numeric($userId) || !is_numeric($year) || !is_numeric($month) || !is_bool($order_desc)) return false;
 112 
 113                 $time_start = mktime(0, 0, 0, $month, 1, $year);                        //00:00 at first day of month
 114                 $time_end   = mktime(23, 59, 59, $month+1, 0, $year);   //23:59 at last day of month
 115 
 116                 $q  = 'SELECT * FROM tblBlogs ';
 117                 $q .= 'WHERE userId='.$userId.' AND deletedBy=0 ';
 118                 $q .= 'AND timeCreated BETWEEN "'.sql_datetime($time_start).'" AND "'.sql_datetime($time_end).'"';
 119                 if ($order_desc === true) {
 120                         $q .= ' ORDER BY timeCreated DESC';
 121                 } else {
 122                         $q .= ' ORDER BY timeCreated ASC';
 123                 }
 124                 return $db->getArray($q);
 125         }

 255         function showUserBlogs($_userid_name = '')
 256         {
 257                 global $session;
 258 
 259                 if ($_userid_name && isset($_GET[$_userid_name]) && is_numeric($_GET[$_userid_name])) {
 260                         $userId = $_GET[$_userid_name];
 261                         echo 'Blogs:'.getUserName($userId).'<br/>';
 262                 } else {
 263                         $userId = $session->id;
 264                         echo 'Your blogs:<br/>';
 265                 }
 266 
 267                 $list = getBlogsByCategory($userId);
 268 
 269                 $shown_category = false;
 270                 foreach ($list as $row) {
 271                         if ($row['categoryId'] != $shown_category) {
 272                                 if (!$row['categoryName']) echo '<div class="blogs_cathead">Uncategorized</div>';
 273                                 else echo '<div class="blogs_cathead">'.$row['categoryName'].'</div>';
 274                                 $shown_category = $row['categoryId'];
 275                         }
 276                         echo '<div class="X">';
 277                         echo '<a href="blog.php?Blog:'.$row['blogId'].'">'.$row['blogTitle'].'</a><br/>';
 278                         echo $row['timeCreated'];
