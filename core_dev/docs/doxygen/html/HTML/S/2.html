 120         function handleSessionActions()
 121         {
 122                 global $db, $config;
 123 
 124                 //force session handling to be skipped to disallow automatic requests from keeping a user "logged in"
 125                 if (!empty($config['no_session'])) return;
 126 
 127                 if (!$this->id && !empty($_POST['register_usr']) && !empty($_POST['register_pwd']) && !empty($_POST['register_pwd2'])) {
 128                         $check = $this->registerUser($_POST['register_usr'], $_POST['register_pwd'], $_POST['register_pwd2']);
 129                         if (!is_numeric($check)) {
 130                                 echo 'Registration failed: '.$check;
 131                                 die;
 132                         }
 133                         $this->logIn($_POST['register_usr'], $_POST['register_pwd']);
 134                 }
 135 
 136                 //Shows login form if 'login' variable is set
 137                 if (!$this->id && isset($_GET['login'])) {
 138                         $session = &$this;      //Required, else any use of $session in header/footer will be undefined references
 139                         require('design_head.php');     //fixme: hur kan jag slippa detta
 140                         $this->showLoginForm();
 141                         require('design_foot.php');
 142                         die;
 143                 }
 144 
 145                 //Check for login request, POST to any page with 'login_usr' & 'login_pwd' variables set to log in
 146                 if (!$this->id) {
 147                         if (!empty($_POST['login_usr']) && !empty($_POST['login_pwd']) && $this->logIn($_POST['login_usr'], $_POST['login_pwd'])) {
 148                                 header('Location: '.$config['web_root'].$this->start_page);
 149                                 die;
 150                         }
 151                 }
 152 
 153                 //Logged in: Check for a logout request. Send GET parameter 'logout' to any page to log out
 154                 if (isset($_GET['logout'])) {
 155                         $this->logOut();
 156                         header('Location: '.$config['web_root'].$this->start_page);
 157                         die;
 158                 }
 159 
 160                 //Logged in: Check if client ip has changed since last request, if so - log user out to avoid session hijacking
 161                 if ($this->check_ip && $this->ip && ($this->ip != IPv4_to_GeoIP($_SERVER['REMOTE_ADDR']))) {
 162                         $this->error = 'Client IP changed';
 163                         $this->log('Client IP changed! Old IP: '.GeoIP_to_IPv4($this->ip).', current: '.GeoIP_to_IPv4($_SERVER['REMOTE_ADDR']), LOGLEVEL_ERROR);
 164                         $this->logOut();
 165                 }
 166 
 167                 //Logged in: Check user activity - log out inactive user
 168                 if ($this->lastActive < (time()-$this->timeout)) {
 169                         $this->error = 'Inactivity timeout';
 170                         $this->log('Session timed out after '.(time()-$this->lastActive).' (timeout is '.($this->timeout).')', LOGLEVEL_NOTICE);
 171                         $this->logOut();
 172                 }
 173 
 174                 //Logged in: Check if client user agent string changed, after active check to avoid useragent change log on auto browser upgrade (Firefox)
 175                 if ($this->check_useragent && $this->user_agent && ($this->user_agent != $_SERVER['HTTP_USER_AGENT'])) {
 176                         $this->error = 'Client user agent string changed';
 177                         $this->log('Client user agent string changed from "'.$this->user_agent.'" to "'.$_SERVER['HTTP_USER_AGENT'].'"', LOGLEVEL_ERROR);
 178                         $this->logOut();
 179                 }
 180 
 181                 if (!$this->id) return;
 182 
 183                 //Update last active timestamp
 184                 $db->query('UPDATE tblUsers SET timeLastActive=NOW() WHERE userId='.$this->id);
 185                 $this->lastActive = time();
 186         }

 201         function registerUser($username, $password1, $password2, $userMode = 0)
 202         {
 203                 global $db, $config;
 204                 if (!is_numeric($userMode)) return false;
 205 
 206                 if ($username != trim($username)) return 'Username contains invalid spaces';
 207 
 208                 if (($db->escape($username) != $username) || ($db->escape($password1) != $password1)) {
 209                         //if someone tries to enter ' or " etc as username/password letters
 210                         //with this check, we dont need to encode the strings for use in sql query
 211                         return 'Username or password contains invalid characters';
 212                 }
 213 
 214                 if (strlen($username) < 3) return 'Username must be at least 3 characters long';
 215                 if (strlen($password1) < 4) return 'Password must be at least 4 characters long';
 216 
 217                 if ($password1 != $password2) return 'The passwords doesnt match';
 218 
 219                 if ($this->reserved_usercheck && isReservedUsername($username)) return 'Username is not allowed';
 220 
 221                 if (getUsersCnt()) {
 222                         $q = 'SELECT userId FROM tblUsers WHERE userName="'.$username.'"';
 223                         $checkId = $db->getOneItem($q);
 224                         if ($checkId) return 'Username already exists';
 225                 } else {
 226                         //No users exists, give this user superadmin status
 227                         $userMode = 2;
 228                 }
 229 
 230                 $q = 'INSERT INTO tblUsers SET userName="'.$username.'",userPass="'.sha1( sha1($this->sha1_key).sha1($password1) ).'",userMode='.$userMode.',timeCreated=NOW()';
 231                 $newUserId = $db->insert($q);
 232 
 233                 $this->log('User <b>'.$username.'</b> created');
 234 
 235                 //Stores the additional data from the userdata fields that's required at registration
 236                 if ($this->userdata) {
 237                         handleRequiredUserdataFields($newUserId);
 238                 }
 239 
 240                 return $newUserId;
 241         }

 243         function logIn($username, $password)
 244         {
 245                 global $db;
 246 
 247                 $enc_username = $db->escape($username);
 248                 $enc_password = sha1( sha1($this->sha1_key).sha1($password) );
 249 
 250                 $q = 'SELECT * FROM tblUsers WHERE userName="'.$enc_username.'" AND userPass="'.$enc_password.'"';
 251                 $data = $db->getOneRow($q);
 252                 if (!$data) {
 253                         $this->error = 'Login failed';
 254                         $this->log('Failed login attempt: username '.$enc_username, LOGLEVEL_WARNING);
 255                         return false;
 256                 }
 257 
 258                 if ($data['userMode'] != 2 && !$this->allow_login) {
 259                         $this->error = 'Logins currently not allowed.';
 260                         return false;
 261                 }
 262                 
 263                 $this->error = '';
 264                 $this->username = $data['userName'];
 265                 $this->id = $data['userId'];
 266                 $this->mode = $data['userMode'];                //0=normal user. 1=admin, 2=super admin
 267                 $this->lastActive = time();
 268 
 269                 if ($this->mode >= 1) $this->isAdmin = 1;
 270                 if ($this->mode >= 2) $this->isSuperAdmin = 1;
 271                 
 272                 //Update last login time
 273                 $db->query('UPDATE tblUsers SET timeLastLogin=NOW(), timeLastActive=NOW() WHERE userId='.$this->id);
 274                 $db->insert('INSERT INTO tblLogins SET timeCreated=NOW(), userId='.$this->id.', IP='.$this->ip.', userAgent="'.$db->escape($_SERVER['HTTP_USER_AGENT']).'"');
 275 
 276                 /* Read in current users settings */
 277                 if ($this->allow_themes) {
 278                         $this->theme = loadUserdataSetting($this->id, 'Theme', $this->default_theme);
 279                 }
 280 
 281                 $this->log('User logged in', LOGLEVEL_NOTICE);
 282                 return true;
 283         }
