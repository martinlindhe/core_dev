  52         function checkStopword($text, $_type)
  53         {
  54                 global $db;
  55 
  56                 if (!is_numeric($_type)) return false;
  57 
  58                 /* Removes non-letters */
  59                 $text = str_replace("\n", ' ', $text);
  60                 $text = str_replace("\r", ' ', $text);
  61                 $text = str_replace('.', '', $text);
  62                 $text = str_replace(',', '', $text);
  63                 $text = str_replace('!', '', $text);
  64                 $text = str_replace('?', '', $text);
  65                 $text = str_replace('(', '', $text);
  66                 $text = str_replace(')', '', $text);
  67 
  68                 while (1) {
  69                         $newtext = str_replace('  ', ' ', $text);
  70                         if ($newtext == $text) break;
  71                         $text = $newtext;
  72                 }
  73                 $text = $db->escape(trim($text));
  74 
  75                 $list = explode(' ', $text);
  76                 $list = array_unique($list);
  77                 $list = array_values($list);
  78 
  79                 $q = 'SELECT wordText,wordMatch FROM tblStopwords WHERE wordType='.$_type;
  80                 $wordlist = $db->getArray($q);
  81 
  82                 foreach ($list as $word) {
  83                         foreach ($wordlist as $stopword) {
  84                                 if ($stopword['wordMatch'] == 1) {
  85                                         /* Match stopword agains the whole input word */
  86                                         if (strtolower($word) == strtolower($stopword['wordText'])) return true;
  87                                 } else {
  88                                         /* Find stopword anywhere inside input word */
  89                                         if (stristr($word, $stopword['wordText'])) return true;
  90                                 }
  91                         }
  92                 }
  93 
  94                 return false;
  95         }

 150         {
 151                 global $db;
 152 
 153                 if (!is_numeric($type) || !is_numeric($full)) return false;
 154                 $word = $db->escape($word);
 155 
 156                 $id = $db->getOneItem('SELECT wordId FROM tblStopwords WHERE wordText="'.$word.'" AND wordType='.$type);
 157                 if ($id) return false;
 158 
 159                 return $db->insert('INSERT INTO tblStopwords SET wordText="'.$word.'",wordType='.$type.',wordMatch='.$full);

 183         //$triggeredBy is only used if $session is not available
 184         function addToModerationQueue($queueType, $itemId, $auto_triggered = false, $triggeredBy = 0)
 185         {
 186                 global $db, $session;
 187 
 188                 if (!is_numeric($itemId) || !is_numeric($queueType) || !is_bool($auto_triggered) || !is_numeric($triggeredBy)) return false;
 189                 if ($auto_triggered != '1') $auto_triggered = 0;
 190 
 191                 $q = 'SELECT queueId FROM tblModeration WHERE itemId='.$itemId.' AND queueType='.$queueType.' AND autoTriggered='.$auto_triggered;
 192                 $queueId = $db->getOneItem($q);
 193                 if ($queueId) return $queueId;
 194 
 195                 if ($session) {         
 196                         $q = 'INSERT INTO tblModeration SET queueType='.$queueType.',itemId='.$itemId.',creatorId='.$session->id.',autoTriggered='.$auto_triggered.',timeCreated=NOW()';
 197                 } else {
 198                         $q = 'INSERT INTO tblModeration SET queueType='.$queueType.',itemId='.$itemId.',creatorId='.$triggeredBy.',autoTriggered='.$auto_triggered.',timeCreated=NOW()';
 199                 }
 200 
 201                 return $db->insert($q);
 202         }

 204         function getModerationQueue($_type, $_sql_limit = '')
 205         {
 206                 global $db;
 207                 if (!is_numeric($_type)) return false;
 208 
 209                 $q  = 'SELECT t1.*,t2.userName AS creatorName FROM tblModeration AS t1 ';
 210                 $q .= 'LEFT JOIN tblUsers AS t2 ON (t1.creatorId=t2.userId) ';
 211                 $q .= 'WHERE t1.moderatedBy=0';
 212                 if ($_type) $q .= ' AND t1.queueType='.$_type;
 213                 $q .= ' ORDER BY t1.timeCreated ASC'.$_sql_limit;
 214 
 215                 return $db->getArray($q);
