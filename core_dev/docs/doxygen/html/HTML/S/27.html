 236         function setUserMode($_id, $_mode)
 237         {
 238                 global $db, $session;
 239                 if (!$session->isSuperAdmin || !is_numeric($_id) || !is_numeric($_mode)) return false;
 240                 
 241                 $q = 'UPDATE tblUsers SET userMode='.$_mode.' WHERE userId='.$_id;
 242                 $db->update($q);
 243 
 244                 if ($_id == $session->id) return true;
 245 
 246                 switch ($_mode) {
 247                         case 0: $msg = $session->username.' has reduced your usermode to normal member.'; break;
 248                         case 1: $msg = $session->username.' has granted you admin rights.'; break;
 249                         case 2: $msg = $session->username.' has granted you super admin rights.'; break;
 250                 }
 251                 sendMessage($_id, 'System message', $msg);
 252 
 253                 $session->log('Changed usermode for '.getUserName($_id).' to '.$_mode);
 254                 return true;
 255         }
