 276         /* Returns a sting like: 4h10m3s */
 277         function shortTimePeriod($seconds)
 278         {
 279                 if (is_float($seconds)) $seconds = round($seconds);
 280                 $retval = '';
 281 
 282                 //years
 283                 $a = date('Y', $seconds) - 1970;
 284                 if ($a==1) $retval = $a.' year, ';
 285                 else if ($a>0) $retval = $a.' years, ';
 286                 $seconds -= (((($a*60)*60)*24)*30)*365;
 287 
 288                 //months
 289                 $a = date('n',$seconds)-1;
 290                 if ($a==1) $retval .= $a.' month, ';
 291                 else if($a>0) $retval .= $a.' months, ';
 292                 $seconds -= ((($a*60)*60)*24)*30;
 293 
 294                 //days
 295                 $a = date('j',$seconds)-1;
 296                 if ($a==1) $retval .= $a.' day, ';
 297                 else if ($a>0) $retval .= $a.' days, ';
 298                 $seconds -= (($a*60)*60)*24;
 299 
 300                 //hours
 301                 $a = date('H',$seconds)-1;
 302                 if ($a>0) $retval .= $a.'h';
 303                 $seconds -= ($a*60)*60;
 304 
 305                 //minutes
 306                 $a = date('i',$seconds)-0; //translate from 09 to 9 quickly ;)
 307                 if ($a>0) $retval .= $a.'m';
 308                 $seconds -= $a*60;
 309 
 310                 //seconds
 311                 $a = date('s',$seconds)-0;
 312                 if ($a>0) $retval .= $a.'s';
 313 
 314                 if (substr($retval, -2) == ', ') $retval = substr($retval, 0, -2);
 315                 if ($retval == '') $retval = '0s';
 316 
 317                 return $retval;
 318         }

 336                         } else {

 338                         }
 339                         $art['body'] = '';
 340 
 341                         return $art;
 342                 }
 343 
 344                 $art['head'] = substr($text, $pos1+strlen('[head]'), $pos2-$pos1-strlen('[/head]')+1);
 345                 $art['head'] = nl2br(trim(strip_tags($art['head'])));
 346 
 347                 $pos1 = strpos($text, '[body]');
 348                 $pos2 = strpos($text, '[/body]');
 349                 if ($pos1 === false || $pos2 === false) return $art;
 350 
 351                 $art['body'] = substr($text, $pos1+strlen('[body]'), $pos2-$pos1-strlen('[/body]')+1);
 352                 $art['body'] = formatUserInputText($art['body']);
 353 
 354                 return $art;
